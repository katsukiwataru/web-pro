# docker を起動する
cd
cd webpro
docker-compose up -d

# MySQL につなぐ
docker ps
docker exec -it <<mysql のコンテナID>> bash
mysql -u root

# MySQL にアプリから接続する用のユーザを作成する
# (mysql user 作成) というキーワードで検索すると出てくる
grant all privileges on *.* to myuser@'%'
    identified by 'password' with grant option;
flush privileges;

# つなげるか試してみる
# web server にログインして、mysql コマンドで接続を試す
docker ps
docker exec -it <<web サーバのコンテナID>> bash
mysql -u myuser -p -h db
<<password と入力>>
test

# PHP からつなげるか試してみる
<?php
    try  {
        // SELECT 文
        $dbh = new PDO('mysql:host=db;dbname=webprodb', 'myuser', 'password');
        $stmt = $dbh->query('SELECT * FROM students');
        while ($row = $stmt->fetch(PDO::FETCH_ASSOC)) {
            print_r($row);
            echo '<br />';
        }
    } catch (PDOException $e) {
        var_dump($e);
        exit;
    }


    try {
        // INSERT 文
        // プリペアドステートメント
        $dbh = new PDO('mysql:host=db;dbname=webprodb', 'myuser', 'password');
        $stmt = $dbh->prepare('INSERT INTO students ("code", "name", "grade") VALUES (:code, :name, :grade)';
        $stmt->execute(array(':code' => 'A17DC000', ':name' => 'YAMADA TARO', ':grade' => 3));
    } catch (PDOException $e) {
        var_dump($e);
        exit;
    }
